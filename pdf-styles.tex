% --- KOMA-Script: Truly Inline, Flush-Left Footnotes ---
% 1. {0pt}: Sets the hanging indent for wrapped lines (0pt = flush left)
% 2. {1em}: Sets the indent for the first line of NEW paragraphs inside one footnote
% 3. Mark: We define it as a left-aligned box of its own natural width
\deffootnote{0pt}{1em}{%
  \makebox[\width][l]{\thefootnotemark.\ }%
}
